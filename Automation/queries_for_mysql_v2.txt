create table PRODUCER_TBL(
	producer_id 	INT NOT NULL,
	name		VARCHAR(50),
	address1	VARCHAR(50),
	address2	VARCHAR(50),
	county		VARCHAR(30),
	city		VARCHAR(30),
	district	VARCHAR(30),
	state		VARCHAR(30),
	region		VARCHAR(30),
	zip		VARCHAR(20),
	country		VARCHAR(30),
	tel		VARCHAR(30),
	PRIMARY KEY(producer_id)
);

create table LOCATION_TBL(
	location_id	VARCHAR(30) NOT NULL,
	producer_id 	INT NOT NULL,
	name		VARCHAR(50),
	address1	VARCHAR(50),
	address2	VARCHAR(50),
	county		VARCHAR(30),
	city		VARCHAR(30),
	district	VARCHAR(30),
	state		VARCHAR(30),
	region		VARCHAR(30),
	zip		VARCHAR(20),
	country		VARCHAR(30),
	location_name	VARCHAR(30),
	tel		VARCHAR(30),
	PRIMARY KEY (location_id,producer_id),
	FOREIGN KEY (producer_id) REFERENCES PRODUCER_TBL(producer_id)
);

create table HERD_TBL(
	herd_id		VARCHAR(30) NOT NULL,
	producer_id	INT	NOT NULL,
	herd_name	VARCHAR(30),
	herd_desc	TEXT,
	PRIMARY KEY (herd_id,producer_id),
	FOREIGN KEY (producer_id) REFERENCES PRODUCER_TBL (producer_id)
);

create table HERD_ASSO_TBL(
	herd_id		VARCHAR(30) NOT NULL,
	producer_id	INT	NOT NULL,
	entry_date	TIMESTAMP	NOT NULL,
	exit_date	TIMESTAMP,
	location_id	VARCHAR(30),
	PRIMARY KEY (herd_id,producer_id,entry_date),
	FOREIGN KEY (producer_id) REFERENCES PRODUCER_TBL (producer_id),
	FOREIGN KEY (herd_id) REFERENCES HERD_TBL (herd_id),
	FOREIGN KEY (location_id) REFERENCES LOCATION_TBL (location_id)
);

create table CATTLE_INFO_TBL(
	herd_id		VARCHAR(30),
	chaps_id	BIGINT NOT NULL AUTO_INCREMENT,
	animal_id		varchar(20),
	birth_date 	DATE,
	breed		VARCHAR(16),
	reg_no		VARCHAR(30),
	reg_name	VARCHAR(30),
	elec_id		VARCHAR(30),
	sire_chaps_id	BIGINT,
	sire_id		varchar(20),
	dam_chaps_id	BIGINT,
	dam_id		varchar(20),
	cow_age		TINYINT,
	sex		VARCHAR(1),	
	birth_weight	FLOAT,
	calving_ease	TINYINT,
	state		VARCHAR(2),
	sex_date	DATE,
	lot_no 		varchar(16),
	picture 	MEDIUMBLOB,
	PRIMARY KEY (chaps_id)
);

create table OWNERS_TBL(
	chaps_id	BIGINT 	NOT NULL,
	start_date	DATE NOT NULL,
	exit_date	DATE,
	herd_id		VARCHAR(30),
	producer_id	INT,
	entry_code	VARCHAR(2),
	exit_code	VARCHAR(2),
	active_flag 	ENUM('A','C','P'),
	animal_source ENUM('S','P'),	
	PRIMARY KEY (chaps_id,start_date),
	FOREIGN KEY (chaps_id) REFERENCES CATTLE_INFO_TBL (chaps_id)ON UPDATE CASCADE ON DELETE CASCADE,
	FOREIGN KEY (producer_id) REFERENCES HERD_ASSO_TBL (producer_id)ON UPDATE CASCADE ON DELETE CASCADE,
	FOREIGN KEY (herd_id) REFERENCES HERD_ASSO_TBL (herd_id) ON UPDATE CASCADE ON DELETE CASCADE
);
